273  return std::any_of(
274  A->excludes().begin(), A->excludes().end(), [A, B](const auto *E) {

458 bool FeatureModelSxfmParser::parseFeatureTree(xmlNode *FeatureTree) {
459  // Split the lines of the feature tree by new lines
460  {
461  std::stringstream Ss(reinterpret_cast<const char *>(
462  UniqueXmlChar(xmlNodeGetContent(FeatureTree), xmlFree).get()));
463  std::string To;
464  std::string Name;
465  bool Opt;
466  bool IsRoot;
467  int LastIndentationLevel = -1;
468  int RootIndentation = -1;
469  int OrGroupCounter = 0;
470  std::map<int, std::string> IndentationToParentMapping;
471 
472  // This map is used for the or group mapping
473  // Each entry represents an or group as a tuple where the first value is
474  // the name of the parent, the second is the relationship kind, and the
475  // third a vector consisting of the name of the children
476  std::map<int, std::tuple<std::string, Relationship::RelationshipKind>>
477  OrGroupMapping;
478 
479  if (FeatureTree == nullptr) {
480  llvm::errs() << "Failed to read in feature tree. Is it empty?\n";
481  return false;
482  }
483 
484  while (std::getline(Ss, To)) {
485  Opt = false;
486  IsRoot = false;
487 
488  if (To.empty() || std::all_of(To.begin(), To.end(), isspace)) {
489  continue;
490  }
491 
492  // For every line, count the indentation
493  // not more than 1 additional indentations are allowed to the original one
494  // However, we may have arbitrarily less indentations
495  if ((To.find(':', 0)) == std::string::npos) {
496  llvm::errs() << "Colon is missing in line" << To << "\n";
497  return false;
498  }
499 
500  llvm::StringRef ToStringRef(To);
501  llvm::StringRef IndentationString =
502  ToStringRef.substr(0, ToStringRef.find(':'));
503  int CurrentIndentationLevel =
504  checkedNarrowingSignConversion(IndentationString.count(Indentation));
505  int Diff = CurrentIndentationLevel - LastIndentationLevel;
506 
507  // Remember the root indentation for later checks
508  if (LastIndentationLevel == -1) {
509  RootIndentation = CurrentIndentationLevel;
510  }
511 
512  if ((LastIndentationLevel != -1) && Diff > 1) {
513  llvm::errs() << "Indentation error in feature tree in line " << To
514  << "\n";
515  return false;
516  }
517 
518  // Move pointer to first character after indentation
519  // The first character has to be a colon followed by the type of
520  // the feature (m for mandatory, o for optional, a for alternative)
521  std::string::size_type Pos =
522  CurrentIndentationLevel * Indentation.length() + 2;
523  llvm::Optional<std::tuple<int, int>> Cardinalities;
524 
525  switch (To.at(Pos - 1)) {
526  case 'r':
527  IsRoot = true;
528  break;
529  case 'm':
530  break;
531  case 'o':
532  // Code for optional
533  Opt = true;
534  break;
535  case 'g':
536  // Code for an or group with different cardinalities
537  Opt = false;
538  // Extract the cardinality
539  Cardinalities = extractCardinality(To);
540  if (!Cardinalities.hasValue()) {
541  return false;
542  }
543  break;
544  case ' ':
545  // Code for alternative child
546  Pos--;
547  break;
548  default:
549  llvm::errs()
550  << "Wrong indentation or unsupported type of configuration option:'"
551  << To << "'\n";
552  return false;
553  }
554  // Extract the name
555  Name =
556  ToStringRef.substr(Pos + 1, ToStringRef.find(' ', Pos + 1) - Pos - 1)
557  .str();
558 
559  // Remove the cardinality
560  if (Name.find_first_of('[') != std::string::npos) {
561  Name = ToStringRef
562  .substr(Pos + 1, ToStringRef.find('[', Pos + 1) - Pos - 1)
563  .str();
564 
565  if (Name.empty()) {
566  // In this case, the name could also be after the cardinality.
567  // According to the examples provided by S.P.L.O.T., this is a valid
568  // format.
569  auto Remainder =
570  ToStringRef.substr(Pos + 1, ToStringRef.size() - Pos - 1);
571  auto Tokens = Remainder.split(' ');
572  if (!Tokens.second.empty()) {
573  Name = Tokens.second.str();
574  }
575  }
576  }
577 
578  // Note that we ignore the ID and use the name of the feature
579  // as unique identifier.
580  if (Name.find_first_of('(') != std::string::npos) {
581  Name = Name.substr(0, Name.find('(', 0));
582  }
583 
584  // Search for identifier
585  std::regex Regex(R"(\([a-zA-Z\d_]*\))");
586  std::string CurrentLine = ToStringRef.str();
587  for (std::sregex_iterator Match = std::sregex_iterator(
588  CurrentLine.begin(), CurrentLine.end(), Regex);
589  Match != std::sregex_iterator(); Match++) {
590  std::string Identifier = (*Match).str();
591  Identifier = Identifier.substr(1, Identifier.size() - 2);
592  IdentifierMap[Identifier] = Name;
593  if (Identifier != Name && Name.find(Identifier) != std::string::npos) {
594  llvm::errs() << "Name must not contain ID:'" << Name << "'\n";
595  return false;
596  }
597  }
598 
599  // If there is no name, provide an artificial one
600  if (Name.empty()) {
601  OrGroupCounter++;
602  Name = "group_" + std::to_string(OrGroupCounter);
603  }
604 
605  // Create the feature
606  if (IsRoot) {
607  FMB.makeRoot(Name);
608  } else {
609  FMB.makeFeature<BinaryFeature>(Name, Opt);
610  }
611  IndentationToParentMapping[CurrentIndentationLevel] = Name;
612 
613  // Add parent from the upper indentation level if there is one
614  if (LastIndentationLevel != -1 &&
615  CurrentIndentationLevel == RootIndentation) {
616  llvm::errs() << "Only one feature can be root and have the same "
617  "indentation as root.\n";
618  return false;
619  }
620 
621  if (LastIndentationLevel != -1) {
622  auto Parent =
623  IndentationToParentMapping.find(CurrentIndentationLevel - 1);
624  assert(Parent != IndentationToParentMapping.end());
625  FMB.addEdge(Parent->second, Name);
626  }
627 
628  // Add the or group to the feature model if it is completely parsed
629  auto OrGroup = OrGroupMapping.find(CurrentIndentationLevel);
630  if (OrGroup != OrGroupMapping.end()) {
631  FMB.emplaceRelationship(std::get<1>(OrGroup->second),
632  std::get<0>(OrGroup->second));
633  OrGroupMapping.erase(CurrentIndentationLevel);
634  }
635 
636  // Remember the new or group parent if there is one
637  if (Cardinalities.hasValue()) {
638  Relationship::RelationshipKind GroupKind =
639  Relationship::RelationshipKind::RK_ALTERNATIVE;
640  if (std::get<1>(Cardinalities.getValue()) == SxfmConstants::WILDCARD) {
641  GroupKind = Relationship::RelationshipKind::RK_OR;
642  }
643  OrGroupMapping[CurrentIndentationLevel] =
644  std::tuple<std::string, Relationship::RelationshipKind>(Name,
645  GroupKind);
646  }
647 
648  LastIndentationLevel = CurrentIndentationLevel;
649  }
650 
651  // Add the remaining or groups
652  for (auto &OrGroup : OrGroupMapping) {
653  FMB.emplaceRelationship(std::get<1>(OrGroup.second),
654  std::get<0>(OrGroup.second));
655  }
656  }
657 
658  return true;
659 }

661 bool FeatureModelSxfmParser::parseConstraints(xmlNode *Constraints) {
662  if (Constraints == nullptr) {
663  llvm::errs()
664  << "Failed to read in constraints. Is the constraint field empty?\n";
665  return false;
666  }
667 
668  std::stringstream Ss(reinterpret_cast<const char *>(
669  UniqueXmlChar(xmlNodeGetContent(Constraints), xmlFree).get()));
670  std::string To;
671 
672  // Prepare the identifiers for later replacement
673  std::vector<std::string> Keys;
674  Keys.reserve(IdentifierMap.size());
675  std::for_each(IdentifierMap.begin(), IdentifierMap.end(),
676  [&Keys](const auto &Entry) { Keys.push_back(Entry.first); });
677  std::sort(Keys.begin(), Keys.end(), std::greater<std::string>{});
678 
679  while (std::getline(Ss, To)) {
680  // Ignore if a line is empty
681  if (To.empty() || std::all_of(To.begin(), To.end(), isspace)) {
682  continue;
683  }
684 
685  // We ignore the indentation in the beginning of each line since the format
686  // does not force it
687  llvm::StringRef ToStringRef(To);
688 
689  if (!ToStringRef.contains(':')) {
690  llvm::errs() << "Failed to read in a constraint since it does not "
691  "contain a colon.";
692  llvm::errs() << "This violates the SXFM format.";
693  return false;
694  }
695 
696  auto Pos = ToStringRef.find(':');
697  ToStringRef = ToStringRef.substr(Pos + 1, ToStringRef.size() - Pos - 1);
698  std::string CnfFormula = trim(ToStringRef);
699 
700  // In the following lines, we replace all identifiers by the real feature
701  // name
702  for (auto &Key : Keys) {
703  std::string Value = IdentifierMap[Key];
704  if (Value == Key) {
705  continue;
706  }
707  auto FoundPos = CnfFormula.find(Key);
708  while (FoundPos != std::string::npos) {
709  CnfFormula = CnfFormula.replace(FoundPos, Key.length(), Value);
710  FoundPos = CnfFormula.find(Key);
711  }
712  }
713 
714  // Replace 'or' by '|' and 'and' by '&'
715  std::unordered_map<std::string, std::string> SpecialKeys;
716  SpecialKeys[" or "] = " | ";
717  SpecialKeys[" and "] = " & ";
718  for (auto &Key : SpecialKeys) {
719  auto FoundPos = CnfFormula.find(Key.first);
720  while (FoundPos != std::string::npos) {
721  CnfFormula =
722  CnfFormula.replace(FoundPos, Key.first.length(), Key.second);
723  FoundPos = CnfFormula.find(Key.first);
724  }
725  }
726 
727  if (auto Constraint = ConstraintParser(CnfFormula).buildConstraint()) {
728  FMB.addConstraint(std::move(Constraint));
729  }
730  }
731 
732  return true;
733 }

735 llvm::Optional<std::tuple<int, int>> FeatureModelSxfmParser::extractCardinality(
736  llvm::StringRef StringToExtractFrom) {
737  llvm::Optional<int> MinCardinality;
738  llvm::Optional<int> MaxCardinality;
739 
740  // Search for the first occurrence of '['; then read in the min cardinality
741  // until the comma. Afterwards, read in the max cardinality until ']'
742  std::string::size_type Pos = StringToExtractFrom.find_first_of('[');
743  if (Pos == std::string::npos) {
744  llvm::errs() << "No cardinality given in or group!\n";
745  return {};
746  }
747  llvm::StringRef CardinalityString(StringToExtractFrom);
748  size_t CommaPos = CardinalityString.find(',', Pos + 1);
749  MinCardinality = parseCardinality(
750  CardinalityString.substr(Pos + 1, CommaPos - Pos - 1).str());
751  Pos = CommaPos;
752  MaxCardinality = parseCardinality(
753  CardinalityString
754  .substr(Pos + 1, CardinalityString.find(']', Pos + 1) - Pos - 1)
755  .str());
756 
757  if (!MinCardinality.hasValue() || !MaxCardinality.hasValue()) {
758  llvm::errs() << "No parsable cardinality!\n";
759  return {};
760  }
761 
762  if (MinCardinality.getValue() != 1 ||
763  (MaxCardinality.getValue() != 1 &&
764  MaxCardinality.getValue() != SxfmConstants::WILDCARD)) {
765  llvm::errs() << "Cardinality unsupported. We support cardinalities [1,1] "
766  "(alternative) or [1, *] (or group).\n";
767  return {};
768  }
769 
770  return {std::tuple<int, int>{MinCardinality.getValue(),
771  MaxCardinality.getValue()}};
772 }

774 llvm::Optional<int>
775 FeatureModelSxfmParser::parseCardinality(llvm::StringRef CardinalityString) {
776  llvm::Optional<int> Result;
777  if (CardinalityString == "*") {
778  // We use -1 as our magic integer to indicate that the cardinality is a
779  // wildcard.
780  Result = SxfmConstants::WILDCARD;
781  } else {
782  // Convert the string into an integer in a safe way
783  int CardinalityValue;
784  if (!llvm::to_integer(CardinalityString, CardinalityValue, 10)) {
785  llvm::errs() << llvm::formatv(
786  "The cardinality: '{0}' was not an {1}-bit integer.\n",
787  CardinalityString, sizeof(CardinalityValue) * 8);
788  } else {
789  Result = CardinalityValue;
790  }
791  }
792 
793  return Result;
794 }
