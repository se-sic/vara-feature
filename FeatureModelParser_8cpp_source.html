353 /// Decide whether feature A excludes B. Beware that this method only detects
354 /// very simple trees with binary excludes.
355 bool detectExclude(const Feature *A, const Feature *B) {
356  return std::any_of(
357  A->excludes().begin(), A->excludes().end(), [A, B](const auto *E) {
358  if (const auto *LHS =
359  llvm::dyn_cast<PrimaryFeatureConstraint>(E->getLeftOperand())) {
360  if (const auto *RHS = llvm::dyn_cast<PrimaryFeatureConstraint>(
361  E->getRightOperand())) {
362  return (LHS->getFeature() &&
363  LHS->getFeature()->getName() == A->getName() &&
364  RHS->getFeature() &&
365  RHS->getFeature()->getName() == B->getName());
366  }
367  }
368  return false;
369  });

542 bool FeatureModelSxfmParser::parseFeatureTree(xmlNode *FeatureTree) {
543  // Split the lines of the feature tree by new lines
544  {
545  std::stringstream Ss(reinterpret_cast<const char *>(
546  UniqueXmlChar(xmlNodeGetContent(FeatureTree), xmlFree).get()));
547  std::string To;
548  std::string Name;
549  bool Opt;
550  bool IsRoot;
551  int LastIndentationLevel = -1;
552  int RootIndentation = -1;
553  int OrGroupCounter = 0;
554  std::unordered_map<int, std::string> IndentationToParentMapping;
555 
556  // This map is used for the or group mapping
557  // Each entry represents an or group as a tuple where the first value is
558  // the name of the parent, the second is the relationship kind, and the
559  // third a vector consisting of the name of the children
560  std::unordered_map<int,
561  std::tuple<std::string, Relationship::RelationshipKind>>
562  OrGroupMapping;
563 
564  if (FeatureTree == nullptr) {
565  llvm::errs() << "Failed to read in feature tree. Is it empty?\n";
566  return false;
567  }
568 
569  while (std::getline(Ss, To)) {
570  Opt = false;
571  IsRoot = false;
572 
573  if (To.empty() || std::all_of(To.begin(), To.end(), isspace)) {
574  continue;
575  }
576 
577  // For every line, count the indentation
578  // not more than 1 additional indentations are allowed to the original one
579  // However, we may have arbitrarily less indentations
580  if ((To.find(':', 0)) == std::string::npos) {
581  llvm::errs() << "Colon is missing in line" << To << "\n";
582  return false;
583  }
584 
585  llvm::StringRef ToStringRef(To);
586  llvm::StringRef IndentationString =
587  ToStringRef.substr(0, ToStringRef.find(':'));
588  int CurrentIndentationLevel =
589  checkedNarrowingSignConversion(IndentationString.count(Indentation));
590  int Diff = CurrentIndentationLevel - LastIndentationLevel;
591 
592  // Remember the root indentation for later checks
593  if (LastIndentationLevel == -1) {
594  RootIndentation = CurrentIndentationLevel;
595  }
596 
597  if ((LastIndentationLevel != -1) && Diff > 1) {
598  llvm::errs() << "Indentation error in feature tree in line " << To
599  << "\n";
600  return false;
601  }
602 
603  // Move pointer to first character after indentation
604  // The first character has to be a colon followed by the type of
605  // the feature (m for mandatory, o for optional, a for alternative)
606  std::string::size_type Pos =
607  CurrentIndentationLevel * Indentation.length() + 2;
608  std::optional<std::tuple<int, int>> Cardinalities;
609 
610  switch (To.at(Pos - 1)) {
611  case 'r':
612  IsRoot = true;
613  break;
614  case 'm':
615  break;
616  case 'o':
617  // Code for optional
618  Opt = true;
619  break;
620  case 'g':
621  // Code for an or group with different cardinalities
622  Opt = false;
623  // Extract the cardinality
624  Cardinalities = extractCardinality(To);
625  if (!Cardinalities.has_value()) {
626  return false;
627  }
628  break;
629  case ' ':
630  // Code for alternative child
631  Pos--;
632  break;
633  default:
634  llvm::errs()
635  << "Wrong indentation or unsupported type of configuration option:'"
636  << To << "'\n";
637  return false;
638  }
639  // Extract the name
640  Name =
641  ToStringRef.substr(Pos + 1, ToStringRef.find(' ', Pos + 1) - Pos - 1)
642  .str();
643 
644  // Remove the cardinality
645  if (Name.find_first_of('[') != std::string::npos) {
646  Name = ToStringRef
647  .substr(Pos + 1, ToStringRef.find('[', Pos + 1) - Pos - 1)
648  .str();
649 
650  if (Name.empty()) {
651  // In this case, the name could also be after the cardinality.
652  // According to the examples provided by S.P.L.O.T., this is a valid
653  // format.
654  auto Remainder =
655  ToStringRef.substr(Pos + 1, ToStringRef.size() - Pos - 1);
656  auto Tokens = Remainder.split(' ');
657  if (!Tokens.second.empty()) {
658  Name = Tokens.second.str();
659  }
660  }
661  }
662 
663  // Note that we ignore the ID and use the name of the feature
664  // as unique identifier.
665  if (Name.find_first_of('(') != std::string::npos) {
666  Name = Name.substr(0, Name.find('(', 0));
667  }
668 
669  // Search for identifier
670  std::regex Regex(R"(\‍([a-zA-Z\d_]*\))");
671  std::string CurrentLine = ToStringRef.str();
672  for (std::sregex_iterator Match = std::sregex_iterator(
673  CurrentLine.begin(), CurrentLine.end(), Regex);
674  Match != std::sregex_iterator(); Match++) {
675  std::string Identifier = (*Match).str();
676  Identifier = Identifier.substr(1, Identifier.size() - 2);
677  IdentifierMap[Identifier] = Name;
678  if (Identifier != Name && Name.find(Identifier) != std::string::npos) {
679  llvm::errs() << "Name must not contain ID:'" << Name << "'\n";
680  return false;
681  }
682  }
683 
684  // If there is no name, provide an artificial one
685  if (Name.empty()) {
686  OrGroupCounter++;
687  Name = "group_" + std::to_string(OrGroupCounter);
688  }
689 
690  // Create the feature
691  if (IsRoot) {
692  FMB.makeRoot(Name);
693  } else {
694  FMB.makeFeature<BinaryFeature>(Name, Opt);
695  }
696  IndentationToParentMapping[CurrentIndentationLevel] = Name;
697 
698  // Add parent from the upper indentation level if there is one
699  if (LastIndentationLevel != -1 &&
700  CurrentIndentationLevel == RootIndentation) {
701  llvm::errs() << "Only one feature can be root and have the same "
702  "indentation as root.\n";
703  return false;
704  }
705 
706  if (LastIndentationLevel != -1) {
707  auto Parent =
708  IndentationToParentMapping.find(CurrentIndentationLevel - 1);
709  assert(Parent != IndentationToParentMapping.end());
710  FMB.addEdge(Parent->second, Name);
711  }
712 
713  // Add the or group to the feature model if it is completely parsed
714  auto OrGroup = OrGroupMapping.find(CurrentIndentationLevel);
715  if (OrGroup != OrGroupMapping.end()) {
716  FMB.emplaceRelationship(std::get<1>(OrGroup->second),
717  std::get<0>(OrGroup->second));
718  OrGroupMapping.erase(CurrentIndentationLevel);
719  }
720 
721  // Remember the new or group parent if there is one
722  if (Cardinalities.has_value()) {
723  Relationship::RelationshipKind GroupKind =
724  Relationship::RelationshipKind::RK_ALTERNATIVE;
725  if (std::get<1>(Cardinalities.value()) == SxfmConstants::WILDCARD) {
726  GroupKind = Relationship::RelationshipKind::RK_OR;
727  }
728  OrGroupMapping[CurrentIndentationLevel] =
729  std::tuple<std::string, Relationship::RelationshipKind>(Name,
730  GroupKind);
731  }
732 
733  LastIndentationLevel = CurrentIndentationLevel;
734  }
735 
736  // Add the remaining or groups
737  for (auto &OrGroup : OrGroupMapping) {
738  FMB.emplaceRelationship(std::get<1>(OrGroup.second),
739  std::get<0>(OrGroup.second));
740  }
741  }
742 
743  return true;
744 }

746 bool FeatureModelSxfmParser::parseConstraints(xmlNode *Constraints) {
747  if (Constraints == nullptr) {
748  llvm::errs()
749  << "Failed to read in constraints. Is the constraint field empty?\n";
750  return false;
751  }
752 
753  std::stringstream Ss(reinterpret_cast<const char *>(
754  UniqueXmlChar(xmlNodeGetContent(Constraints), xmlFree).get()));
755  std::string To;
756 
757  // Prepare the identifiers for later replacement
758  std::vector<std::string> Keys;
759  Keys.reserve(IdentifierMap.size());
760  std::for_each(IdentifierMap.begin(), IdentifierMap.end(),
761  [&Keys](const auto &Entry) { Keys.push_back(Entry.first); });
762  std::sort(Keys.begin(), Keys.end(), std::greater<std::string>{});
763 
764  while (std::getline(Ss, To)) {
765  // Ignore if a line is empty
766  if (To.empty() || std::all_of(To.begin(), To.end(), isspace)) {
767  continue;
768  }
769 
770  // We ignore the indentation in the beginning of each line since the format
771  // does not force it
772  llvm::StringRef ToStringRef(To);
773 
774  if (!ToStringRef.contains(':')) {
775  llvm::errs() << "Failed to read in a constraint since it does not "
776  "contain a colon.";
777  llvm::errs() << "This violates the SXFM format.";
778  return false;
779  }
780 
781  auto Pos = ToStringRef.find(':');
782  ToStringRef = ToStringRef.substr(Pos + 1, ToStringRef.size() - Pos - 1);
783  std::string CnfFormula = trim(ToStringRef);
784 
785  // In the following lines, we replace all identifiers by the real feature
786  // name
787  for (auto &Key : Keys) {
788  std::string Value = IdentifierMap[Key];
789  if (Value == Key) {
790  continue;
791  }
792  auto FoundPos = CnfFormula.find(Key);
793  while (FoundPos != std::string::npos) {
794  CnfFormula = CnfFormula.replace(FoundPos, Key.length(), Value);
795  FoundPos = CnfFormula.find(Key);
796  }
797  }
798 
799  // Replace 'or' by '|' and 'and' by '&'
800  std::unordered_map<std::string, std::string> SpecialKeys;
801  SpecialKeys[" or "] = " | ";
802  SpecialKeys[" and "] = " & ";
803  for (auto &Key : SpecialKeys) {
804  auto FoundPos = CnfFormula.find(Key.first);
805  while (FoundPos != std::string::npos) {
806  CnfFormula =
807  CnfFormula.replace(FoundPos, Key.first.length(), Key.second);
808  FoundPos = CnfFormula.find(Key.first);
809  }
810  }
811 
812  if (auto Constraint = ConstraintParser(CnfFormula).buildConstraint()) {
813  FMB.addConstraint(std::make_unique<FeatureModel::BooleanConstraint>(
814  std::move(Constraint)));
815  }
816  }
817 
818  return true;
819 }

821 std::optional<std::tuple<int, int>> FeatureModelSxfmParser::extractCardinality(
822  llvm::StringRef StringToExtractFrom) {
823  std::optional<int> MinCardinality;
824  std::optional<int> MaxCardinality;
825 
826  // Search for the first occurrence of '['; then read in the min cardinality
827  // until the comma. Afterwards, read in the max cardinality until ']'
828  std::string::size_type Pos = StringToExtractFrom.find_first_of('[');
829  if (Pos == std::string::npos) {
830  llvm::errs() << "No cardinality given in or group!\n";
831  return {};
832  }
833  llvm::StringRef CardinalityString(StringToExtractFrom);
834  size_t CommaPos = CardinalityString.find(',', Pos + 1);
835  MinCardinality = parseCardinality(
836  CardinalityString.substr(Pos + 1, CommaPos - Pos - 1).str());
837  Pos = CommaPos;
838  MaxCardinality = parseCardinality(
839  CardinalityString
840  .substr(Pos + 1, CardinalityString.find(']', Pos + 1) - Pos - 1)
841  .str());
842 
843  if (!MinCardinality.has_value() || !MaxCardinality.has_value()) {
844  llvm::errs() << "No parsable cardinality!\n";
845  return {};
846  }
847 
848  if (MinCardinality.value() != 1 ||
849  (MaxCardinality.value() != 1 &&
850  MaxCardinality.value() != SxfmConstants::WILDCARD)) {
851  llvm::errs() << "Cardinality unsupported. We support cardinalities [1,1] "
852  "(alternative) or [1, *] (or group).\n";
853  return {};
854  }
855 
856  return {std::tuple<int, int>{MinCardinality.value(), MaxCardinality.value()}};
857 }

859 std::optional<int>
860 FeatureModelSxfmParser::parseCardinality(llvm::StringRef CardinalityString) {
861  std::optional<int> Result;
862  if (CardinalityString == "*") {
863  // We use -1 as our magic integer to indicate that the cardinality is a
864  // wildcard.
865  Result = SxfmConstants::WILDCARD;
866  } else {
867  // Convert the string into an integer in a safe way
868  int CardinalityValue;
869  if (!llvm::to_integer(CardinalityString, CardinalityValue, 10)) {
870  llvm::errs() << llvm::formatv(
871  "The cardinality: '{0}' was not an {1}-bit integer.\n",
872  CardinalityString, sizeof(CardinalityValue) * 8);
873  } else {
874  Result = CardinalityValue;
875  }
876  }
877 
878  return Result;
879 }
