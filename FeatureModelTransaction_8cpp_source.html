61 /// Checks if in recursive mode the provided Feature can be deleted or not.
62 ///
63 /// \param FM
64 /// \param FV
65 /// \param OtherFeatures
66 /// \return optional of true, if all transitive children of FV are not part of
67 /// OtherFeatures; optional of false, if at least one transitive child of FV is
68 /// part of OtherFeatures; nullopt, if FV is not part of FM
69 std::optional<bool>
70 canBeDeletedRecursively(FeatureModel &FM, detail::FeatureVariantTy &FV,
71  const std::set<FeatureTreeNode *> &OtherFeatures) {
72  Feature *ActualFeature = getActualFeature(FM, FV);
73  if (ActualFeature == nullptr) { // Feature does not exist in FM
74  return std::nullopt;
75  }
76 
77  std::set<FeatureTreeNode *> Intersection;
78  auto AllChildrenInSubtree = ActualFeature->getChildren();
79  std::set_intersection(AllChildrenInSubtree.begin(),
80  AllChildrenInSubtree.end(), OtherFeatures.begin(),
81  OtherFeatures.end(),
82  std::inserter(Intersection, Intersection.begin()));
83  // if the children are not matching the other features, the intersection is
84  // empty and we are good to go for deletion
85  return std::optional<bool>{Intersection.begin() == Intersection.end()};
86 }

88 std::vector<detail::FeatureVariantTy> removeFeatures(
89  FeatureModel &FM, std::vector<detail::FeatureVariantTy>::iterator Begin,
90  std::vector<detail::FeatureVariantTy>::iterator End, bool Recursive) {
91  // if everything was deleted
92  std::vector<detail::FeatureVariantTy> NotDeletedFeatures;
93  if (Begin == End) {
94  return NotDeletedFeatures;
95  }
96 
97  std::set<FeatureTreeNode *> OtherFeatures;
98  std::transform(
99  Begin, End, std::inserter(OtherFeatures, OtherFeatures.begin()),
100  [&FM](detail::FeatureVariantTy &FV) { return getActualFeature(FM, FV); });
101  // Remove nullptr, in case, on of the specified Features could not be
102  // found in the FeatureModel
103  OtherFeatures.erase(nullptr);
104 
105  // 3 use-cases:
106  // If Feature is Leave --> can be deleted
107  // If Feature is not a Leave and !Recursive --> cannot be deleted
108  // If Recursive --> partition by "youngest" feature in sub-tree --> delete
109  // those in recursive mode
110  auto DeleteIt = std::partition(
111  Begin, End,
112  [&FM, Recursive, &OtherFeatures](detail::FeatureVariantTy &FV) {
113  if (fvIsLeaf(FM, FV).value_or(false)) {
114  return true;
115  }
116  if (!Recursive) {
117  return false;
118  }
119  return canBeDeletedRecursively(FM, FV, OtherFeatures).value_or(false);
120  });
121 
122  // check if something can be deleted --> if not and return non-deletable
123  // features
124  if (DeleteIt == Begin) {
125  return {Begin, End};
126  }
127 
128  // perform actual removal of deletable features
129  auto Trans = FeatureModelModifyTransaction::openTransaction(FM);
130  auto FeatureIt = Begin;
131  while (FeatureIt != DeleteIt) {
132  Trans.removeFeature(*FeatureIt, Recursive);
133  FeatureIt = std::next(FeatureIt);
134  }
135  Trans.commit();
136 
137  // call remove Features on non-leaves
138  return removeFeatures(FM, DeleteIt, End, Recursive);
139 }

172 [[nodiscard]] bool mergeSubtree(FeatureModelCopyTransaction &Trans,
173  FeatureModel const &FM, Feature &F,
174  bool Strict) {
175  // Is there a similar Feature in the original FM
176  if (Feature *CMP = FM.getFeature(F.getName())) {
177  if (compareProperties(*CMP, F, Strict)) {
178  // similar feature, maybe merge locations
179  for (FeatureSourceRange const &FSR : F.getLocations()) {
180  if (std::find(CMP->getLocationsBegin(), CMP->getLocationsEnd(), FSR) ==
181  CMP->getLocationsEnd()) {
182  Trans.addLocation(CMP, FSR);
183  }
184  }
185  } else {
186  return false;
187  }
188  } else {
189  std::unique_ptr<Feature> Copy = featureCopy(F);
190  if (!Copy) {
191  return false;
192  }
193  Trans.addFeature(std::move(Copy), F.getParentFeature());
194  }
195 
196  // copy children if missing
197  for (Feature *Child : F.getChildren<Feature>()) {
198  if (!mergeSubtree(Trans, FM, *Child, Strict)) {
199  return false;
200  }
201  }
202  return true;
203 }

228 [[nodiscard]] bool compareProperties(const Feature &F1, const Feature &F2,
229  bool Strict) {
230  if (F1.getName() != F2.getName()) {
231  return false;
232  }
233  if (F1.getKind() != Feature::FeatureKind::FK_ROOT &&
234  F2.getKind() != Feature::FeatureKind::FK_ROOT) {
235  if (*(F1.getParentFeature()) != *(F2.getParentFeature())) {
236  return false;
237  }
238  }
239  if (!Strict) {
240  // strict merging still requires equal parent names, otherwise we might
241  // introduce implicit constraint due to model structure, e.g., merging group
242  return true;
243  }
244  if (F1.isOptional() != F2.isOptional()) {
245  return false;
246  }
247  if (F1.getKind() != F2.getKind()) {
248  return false;
249  }
250  if (F1.getKind() == Feature::FeatureKind::FK_ROOT) {
251  return true;
252  }
253  if (F1.getParent()->getKind() != F2.getParent()->getKind()) {
254  return false;
255  }
256  if (F1.getKind() == Feature::FeatureKind::FK_BINARY) {
257  return true;
258  }
259  if (const auto *NF1 = llvm::dyn_cast<NumericFeature>(&F1)) {
260  if (const auto *NF2 = llvm::dyn_cast<NumericFeature>(&F2)) {
261  return NF1->getValues() == NF2->getValues();
262  }
263  }
264  return false;
265 }
