44 std::unique_ptr<FeatureModel> FeatureModel::clone() const {
45  FeatureModelBuilder FMB;
46  FMB.setVmName(this->getName().str());
47  FMB.setCommit(this->getCommit().str());
48  FMB.setPath(this->getPath().string());
49 
50  for (const auto &KV : this->Features) {
51  std::vector<FeatureSourceRange> SourceRanges(
52  KV.getValue()->getLocations().begin(),
53  KV.getValue()->getLocations().end());
54 
55  switch (KV.getValue()->getKind()) {
56  case Feature::FeatureKind::FK_UNKNOWN:
57  FMB.makeFeature<Feature>(KV.getValue()->getName().str());
58  break;
59  case Feature::FeatureKind::FK_ROOT:
60  FMB.makeRoot(KV.getValue()->getName().str());
61  break;
62  case Feature::FeatureKind::FK_BINARY:
63  if (auto *F = llvm::dyn_cast<BinaryFeature>(KV.getValue().get()); F) {
64  FMB.makeFeature<BinaryFeature>(F->getName().str(), F->isOptional(),
65  std::move(SourceRanges));
66  }
67  break;
68  case Feature::FeatureKind::FK_NUMERIC:
69  if (auto *F = llvm::dyn_cast<NumericFeature>(KV.getValue().get()); F) {
70  FMB.makeFeature<NumericFeature>(F->getName().str(), F->getValues(),
71  F->isOptional(),
72  std::move(SourceRanges));
73  }
74  break;
75  }
76 
77  if (auto *ParentFeature = KV.getValue()->getParentFeature()) {
78  FMB.addEdge(ParentFeature->getName().str(),
79  KV.getValue()->getName().str());
80  }
81  }
82 
83  for (const auto &Rel : this->Relationships) {
84  if (auto *P = llvm::dyn_cast_or_null<Feature>(Rel->getParent())) {
85  FMB.emplaceRelationship(Rel->getKind(), P->getName().str());
86  }
87  }
88 
89  // We can use recursive cloning on constraints, as we can rely on the
90  // builder to update pointers into the feature model correctly if invoked
91  // afterwards.
92  // TODO(s9latimm): Add unittests for cloned Constraints
93  for (const auto &C : this->booleanConstraints()) {
94  FMB.addConstraint(
95  std::make_unique<FeatureModel::BooleanConstraint>((**C)->clone()));
96  }
97 
98  for (const auto &C : this->nonBooleanConstraints()) {
99  FMB.addConstraint(
100  std::make_unique<FeatureModel::NonBooleanConstraint>((**C)->clone()));
101  }
102 
103  for (const auto &C : this->mixedConstraints()) {
104  FMB.addConstraint(std::make_unique<FeatureModel::MixedConstraint>(
105  C->constraint()->clone(), C->req(), C->exprKind()));
106  }
107 
108  // Build clone of FM which is supposed to work, as the original FM should be
109  // valid.
110  auto FM = FMB.buildFeatureModel();
111  assert(FM);
112  return FM;
113 }
