84  static ResultTy munch(const llvm::StringRef Str) {
85  if (('a' <= Str.front() && Str.front() <= 'z') ||
86  ('A' <= Str.front() && Str.front() <= 'Z')) {
87  return munchIdentifier(Str);
88  }
89  if (('0' <= Str.front() && Str.front() <= '9') || Str.front() == '.') {
90  return munchNumber(Str);
91  }
92 
93  switch (Str.front()) {
94  case EOF:
95  case '\0':
96  return {StepFunctionToken(StepFunctionToken::TokenKind::END_OF_FILE), 1};
97  case ' ':
98  case '\t':
99  case '\r':
100  case '\n':
101  return munchWhitespace(Str);
102  case '+':
103  return {StepFunctionToken(StepFunctionToken::TokenKind::PLUS), 1};
104  case '*':
105  return {StepFunctionToken(StepFunctionToken::TokenKind::STAR), 1};
106  case '^':
107  return {StepFunctionToken(StepFunctionToken::TokenKind::CARET), 1};
108  default:
109  return {StepFunctionToken(StepFunctionToken::TokenKind::ERROR,
110  Str.take_front().str()),
111  1};
112  }
