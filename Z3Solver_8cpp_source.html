8 Z3Solver::addFeature(const feature::Feature &FeatureToAdd,
9  bool IsInAlternativeGroup) {
10  // Check whether the parent feature is already added
11  vara::feature::Feature *Parent = FeatureToAdd.getParentFeature();
12  if (Parent != nullptr && OptionToVariableMapping.find(Parent->getName()) ==
13  OptionToVariableMapping.end()) {
14  return PARENT_NOT_PRESENT;
15  }
16 
17  if (OptionToVariableMapping.find(FeatureToAdd.getName()) !=
18  OptionToVariableMapping.end()) {
19  return ALREADY_PRESENT;
20  }
21 
22  // Add the feature
23  switch (FeatureToAdd.getKind()) {
24  case feature::Feature::FeatureKind::FK_NUMERIC: {
25  const auto *F =
26  llvm::dyn_cast<vara::feature::NumericFeature>(&FeatureToAdd);
27  if (!F) {
28  return NOT_SUPPORTED;
29  }
30  const auto Values = F->getValues();
31  if (std::holds_alternative<vara::feature::NumericFeature::ValueListType>(
32  Values)) {
33  if (auto R = addFeature(
34  F->getName().str(),
35  std::get<vara::feature::NumericFeature::ValueListType>(Values));
36  !R) {
37  return R;
38  }
39  } else {
40  auto Range =
41  std::get<vara::feature::NumericFeature::ValueRangeType>(Values);
42  auto *StepFunction = F->getStepFunction();
43  auto Step = Range.first;
44  std::vector<int64_t> Vals;
45  while (Step <= Range.second) {
46  Vals.insert(Vals.begin(), Step);
47  Step = StepFunction->next(Step);
48  }
49  if (auto R = addFeature(F->getName().str(), Vals); !R) {
50  return R;
51  }
52  }
53  break;
54  }
55  case feature::Feature::FeatureKind::FK_BINARY: {
56  // Add all constraints (i.e., implications, exclusions, parent feature)
57  if (!llvm::isa<vara::feature::BinaryFeature>(&FeatureToAdd)) {
58  return NOT_SUPPORTED;
59  }
60  addFeature(FeatureToAdd.getName().str());

62  *llvm::dyn_cast<vara::feature::BinaryFeature>(&FeatureToAdd),
63  IsInAlternativeGroup);
64  if (!R) {
65  return R;
66  }
67  break;
68  }
69  case feature::Feature::FeatureKind::FK_ROOT:
70  // Add root as a constraint; root is mandatory
71  addFeature(FeatureToAdd.getName().str());

73  OptionToVariableMapping.end() &&
74  "No OptionToVariableMapping was defined for the given feature.");
75  Solver->add(*OptionToVariableMapping[FeatureToAdd.getName()]);
76  break;
77  case feature::Feature::FeatureKind::FK_UNKNOWN:
78  return NOT_SUPPORTED;
79  }
80

82  return Ok();
83 }

229  return UNSAT;

236  const z3::expr Value = M.eval(OptionExpr, true);
237  if (Value.is_bool()) {

241  Expr = Expr || *Iterator->getValue();

287 bool Z3SolverConstraintVisitor::visit(vara::feature::BinaryConstraint *C) {
288 
289  const bool LHS = C->getLeftOperand()->accept(*this);
290  const z3::expr Left = Z3ConstraintExpression;
291  const bool RHS = C->getRightOperand()->accept(*this);
292  if (!LHS || !RHS) {
293  return false;
294  }
295  switch (C->getKind()) {
296  case feature::Constraint::ConstraintKind::CK_ADDITION:
297  Z3ConstraintExpression = Left + Z3ConstraintExpression;
298  break;
299  case feature::Constraint::ConstraintKind::CK_AND:
300  Z3ConstraintExpression = Left && Z3ConstraintExpression;
301  break;
302  case feature::Constraint::ConstraintKind::CK_DIVISION:
303  Z3ConstraintExpression = Left / Z3ConstraintExpression;
304  break;
305  case feature::Constraint::ConstraintKind::CK_EQUAL:
306  Z3ConstraintExpression = Left == Z3ConstraintExpression;
307  break;
308  case feature::Constraint::ConstraintKind::CK_EQUIVALENCE:
309  Z3ConstraintExpression = z3::implies(Left, Z3ConstraintExpression) &&
310  z3::implies(Z3ConstraintExpression, Left);
311  break;
312  case feature::Constraint::ConstraintKind::CK_EXCLUDES:
313  Z3ConstraintExpression = z3::implies(Left, !Z3ConstraintExpression);
314  break;
315  case feature::Constraint::ConstraintKind::CK_GREATER:
316  Z3ConstraintExpression = Left > Z3ConstraintExpression;
317  break;
318  case feature::Constraint::ConstraintKind::CK_GREATER_EQUAL:
319  Z3ConstraintExpression = Left >= Z3ConstraintExpression;
320  break;
321  case feature::Constraint::ConstraintKind::CK_IMPLIES:
322  Z3ConstraintExpression = z3::implies(Left, Z3ConstraintExpression);
323  break;
324  case feature::Constraint::ConstraintKind::CK_LESS:
325  Z3ConstraintExpression = Left < Z3ConstraintExpression;
326  break;
327  case feature::Constraint::ConstraintKind::CK_LESS_EQUAL:
328  Z3ConstraintExpression = Left <= Z3ConstraintExpression;
329  break;
330  case feature::Constraint::ConstraintKind::CK_MULTIPLICATION:
331  Z3ConstraintExpression = Left * Z3ConstraintExpression;
332  break;
333  case feature::Constraint::ConstraintKind::CK_NOT_EQUAL:
334  Z3ConstraintExpression = Left != Z3ConstraintExpression;
335  break;
336  case feature::Constraint::ConstraintKind::CK_OR:
337  Z3ConstraintExpression = Left || Z3ConstraintExpression;
338  break;
339  case feature::Constraint::ConstraintKind::CK_SUBTRACTION:
340  Z3ConstraintExpression = Left - Z3ConstraintExpression;
341  break;
342  case feature::Constraint::ConstraintKind::CK_XOR:
343  Z3ConstraintExpression = z3::implies(Left, !Z3ConstraintExpression) &&
344  z3::implies(Z3ConstraintExpression, !Left);
345  break;
346  default:
347  static_assert(
348  "Unimplemented binary constraint in Z3SolverConstraintVisitor "
349  "detected!");
350  return false;
351  }
352  return true;
353 }
